139 //! Insert an element at the specified position in the list
140 //! \param val the value to be inserted
141 //! \param index position of the list that the element will be inserted on
142  virtual void insert(const T val, const int index) {
143  if (index < 0)
144  throw std::out_of_range("Negative index not allowed.");
145  if (index > size)
146  throw std::out_of_range("Nonexistent index.");
147 
148  // if it's the first element on the list, create the first node
149  if (size == 0) {
150  first = new Node<T>();
151  first->setValue(val);
152 
153  // first and last both point to the same node at the beginning
154  last = first;
155  }
156  // if the insertion is at the beginning,
157  // use the pointer to the first node to
158  // speed things up and replace first too
159  else if (index == 0) {
160  Node<T> *tmp = new Node<T>();
161  tmp->setValue(val);
162  tmp->setNext(first);
163  first->setPrevious(tmp);
164  first = tmp;
165  }
166  // the same applies when an insertion is made in the last node
167  // this is useful for the stack and queue implementations
168  else if (index == size) {
169  last->setNext(new Node<T>());
170  last->getNext()->setValue(val);
171  last->getNext()->setPrevious(last);
172  last = last->getNext();
173  }
174  // if the index is in the middle of the list, it must be traversed
175  else {
176  Node<T> *tmp = getNode(index);
177 
178  // a new node is created and put between
179  // the current node and the next one
180  Node<T> *tmp2 = new Node<T>();
181  tmp2->setValue(val);
182  tmp2->setNext(tmp->getNext());
183  tmp2->setPrevious(tmp);
184 
185  if (tmp->getNext() != NULL)
186  tmp->getNext()->setPrevious(tmp2);
187 
188  tmp->setNext(tmp2);
189  }
190  size++;
191  }
